coords <- st_coordinates(ride_data$geometry)

ride_data$lat <- coords[,2]

# Calculate distance between consecutive points (in feet)
calculate_distances <- function(lon, lat) {
  n <- length(lon)
  dists <- numeric(n-1)
  
  for(i in 1:(n-1)) {
    dists[i] <- distHaversine(
      c(lon[i], lat[i]),
      c(lon[i+1], lat[i+1])
    )
  }
  
  return(c(0, dists))  # First point has 0 distance
}

ride_data$segment_dist <- calculate_distances(ride_data$lon, ride_data$lat)

# Calculate cumulative distance
ride_data$cum_dist <- cumsum(ride_data$segment_dist)

# Convert to miles
ride_data$cum_dist_miles <- ride_data$cum_dist / 1609.34

# Create mile splits
miles <- floor(max(ride_data$cum_dist_miles))
mile_splits <- data.frame(
  mile = 1:miles,
  split_time = numeric(miles),
  avg_hr = numeric(miles),
  elevation_gain = numeric(miles)
)

for(i in 1:miles) {
  # Find points in this mile
  mile_start <- ride_data[which(ride_data$cum_dist_miles >= (i-1) & 
                                ride_data$cum_dist_miles < i), ]
  mile_end <- ride_data[which(ride_data$cum_dist_miles >= i)[1], ]
  
  # Calculate time for this mile (in minutes)
  time_diff <- as.numeric(difftime(mile_end$time[1], mile_start$time[1], units="mins"))
  mile_splits$split_time[i] <- time_diff
  
  
  
  ##------
 
  
  
  # Calculate average heart rate for this mile
  mile_data <- ride_data[which(ride_data$cum_dist_miles >= (i-1) & 
                               ride_data$cum_dist_miles < i), ]
  mile_splits$avg_hr[i] <- mean(as.numeric(mile_data$heart_rate), na.rm=TRUE)
  
  # Calculate elevation gain for this mile
  ele_diffs <- diff(as.numeric(mile_data$ele))
  mile_splits$elevation_gain[i] <- sum(pmax(0, ele_diffs))
}

# Create a pace column (minutes per mile)
mile_splits$pace <- mile_splits$split_time

# Display mile splits table
